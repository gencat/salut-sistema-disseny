import { LitElement, unsafeCSS } from 'lit';
import { property, query, state } from 'lit/decorators.js';
import resetStyles from '../../shared/reset.style.css?inline';
import { lazyLoading } from '../../utils/lazy-loading';
import { applyMask, deleteSeparatorMask } from '../../utils/mask';
import { booleanConverter, booleanType } from '../../utils/property-types';
import inputStyles from './input.style.css?inline';
import { template } from './input.template';
 
export class Input extends LitElement {
	static formAssociated = true;
 
	static get styles() {
		return [unsafeCSS(resetStyles), unsafeCSS(inputStyles)];
	}
 
	private internals: ElementInternals;
 
	constructor() {
		super();
		this.internals = this.attachInternals();
	}
 
	disconnectedCallback() {
		super.disconnectedCallback();
		this._stopHold();
	}
 
	// Inner Elements
	@query('input.dss-input') _input!: HTMLInputElement;
 
	// Native input properties
	@property({ type: String }) label = '';
	@property(booleanType) hideLabel = false; // Turn label into aria-label
	@property({ type: String }) name = '';
	@property({ type: String }) id = '';
	@property({ type: String }) type = 'text';
	@property({ type: String }) placeholder = '';
	@property({ type: String }) value = '';
	@property({ converter: booleanConverter, reflect: true }) disabled = false;
	@property({ converter: booleanConverter, reflect: true }) readonly = false;
	@property({ converter: booleanConverter, reflect: true }) required = false;
	@property({ converter: booleanConverter, reflect: true }) invalid = false;
	@property({ type: Number }) step?: number;
	@property({ type: Number }) min?: number;
	@property({ type: Number }) max?: number;
	@property({ type: Number }) minLength?: number;
	@property({ type: Number }) maxLength?: number;
	@property({ type: String }) pattern?: string;
	@property({ type: String }) inputmode?: string; // 'text', 'numeric', 'tel', etc.
	@property({ type: String }) autocapitalize = 'none'; // 'none', 'sentences', 'words', 'characters'
	@property({ type: String }) autocomplete = 'off';
	@property(booleanType) autocorrect = false;
	@property(booleanType) autofocus = false;
	@property(booleanType) spellcheck = false; // 'true' or 'false'
 
	// Custom properties
	@property({ type: String }) size = 'lg';
	@property({ type: String }) icon?: string;
	@property({ type: String }) helpText?: string;
	@property({ type: String }) maskRegex?: string;
	@property({ type: String }) maskReplace?: string;
	@property({ type: String }) allowedChars?: string;
	@property({ type: String }) unit?: string;
	@property({ type: String }) inputPrefix?: string;
	@property(booleanType) hasActions = false;
 
	private _defaultId = `dss-input-${crypto.randomUUID()}`;
 
	_getEffectiveId() {
		return this.id !== '' ? this.id : this._defaultId;
	}
 
	// States
	@state() _isFocused = false;
	@state() _isTruncated = false;
	@state() _onHoldInterval: number | null = null;
	@state() _previousValue: string | null = null;
 
	updated(changedProps: Map<string, unknown>) {
		if (changedProps.has('value')) {
			this.internals.setFormValue(this.value); // sincroniza con formulario
		}
	}
 
	formDisabledCallback(disabled: boolean) {
		this.disabled = disabled;
	}
 
	formResetCallback() {
		this.value = '';
		this._input.value = '';
		this._lastValue = '';
	}
 
	formStateRestoreCallback(state: string | null) {
		this.value = state ?? '';
		this._input.value = state ?? '';
		this._lastValue = state ?? '';
	}
 
	render() {
		return template(this);
	}
 
	_handleFocusin() {
		this._isFocused = true;
	}
 
	@state() _labelClicked = false;
 
	_handleLabelClick(event: MouseEvent) {
		event.preventDefault();
		this._input.focus();
		this._labelClicked = true;
		setTimeout(() => {
			this._labelClicked = false;
		}, 50); // Flag reset
	}
 
	_handleFocusout() {
		if (this._labelClicked) return;
		this._isFocused = false;
		if (this._lastValue !== this._input.value) {
			this._checkInputOverflow();
			this._lastValue = this._input.value;
		}
	}
 
	_onInput(e: Event) {
		const input = e.target as HTMLInputElement;
		if (this.maskRegex && this.maskReplace) {
			if (this._previousValue && this._previousValue.length > input?.value.length) {
				input.value = deleteSeparatorMask(this._previousValue, input.value, this.maskReplace);
			}
			const masked = applyMask(input.value, this.maskRegex, this.maskReplace, this.allowedChars);
			if (masked !== input.value) {
				input.value = masked;
			}
		}
		this._previousValue = input.value;
		this.value = input.value;
		this._handleValidity();
		this.dispatchEvent(new Event('input', { bubbles: false, composed: true })); // bubbles false to avoid double event in React
		this._emitChange();
	}
 
	_handleValidity() {
		const validity = this._input?.checkValidity();
		this.invalid = !validity;
		this.internals.setValidity(this._input.validity, this._input.validationMessage, this._input);
	}
 
	_roundFloat(value: number): number {
		return Number.parseFloat(value.toPrecision(12));
	}
 
	_stepWithFallback(direction: 1 | -1) {
		const input = this._input;
		if (!this._input) return;
 
		const stepAttribute = input.getAttribute('step')?.toLowerCase();
		const isStepAny = stepAttribute === 'any';
 
		if (isStepAny) {
			const currentValue = Number.isFinite(input.valueAsNumber) ? input.valueAsNumber : 0;
			const min = input.hasAttribute('min') ? Number(input.getAttribute('min')) : null;
			const max = input.hasAttribute('max') ? Number(input.getAttribute('max')) : null;
			const newValue = this._roundFloat(currentValue + direction);
			if (min !== null && newValue < min) return;
			if (max !== null && newValue > max) return;
			this._input.value = String(newValue);
			return;
		}
 
		try {
			if (direction === 1) {
				this._input.stepUp();
			} else {
				this._input.stepDown();
			}
		} catch (error) {
			console.error(error);
		}
	}
 
	_stepUp() {
		this._stepWithFallback(1);
		this.value = this._input.value;
		this._handleValidity();
		this._emitInput();
		this._emitChange();
	}
 
	_stepDown() {
		this._stepWithFallback(-1);
		this.value = this._input.value;
		this._handleValidity();
		this._emitInput();
		this._emitChange();
	}
 
	_onHold(direction: 'up' | 'down') {
		this._onHoldInterval = window.setInterval(() => {
			direction === 'up' ? this._stepUp() : this._stepDown();
		}, 150);
	}
 
	_stopHold() {
		if (this._onHoldInterval !== null) {
			clearInterval(this._onHoldInterval);
			this._onHoldInterval = null;
		}
	}
 
	_emitInput() {
		this.dispatchEvent(new Event('input', { bubbles: true, composed: true }));
	}
 
	_emitChange() {
		this.dispatchEvent(new Event('change', { bubbles: true, composed: true }));
	}
 
	_lastValue: string | null = null;
 
	_handleMouseOver() {
		if (this._input?.value === this._lastValue) return;
		this._checkInputOverflow();
		this._lastValue = this._input?.value || null;
	}
 
	_checkInputOverflow() {
		if (!this._input || !this._input.value) return;
 
		const inputStyle = window.getComputedStyle(this._input);
		const inputFont = `${inputStyle.fontWeight} ${inputStyle.fontSize} ${inputStyle.fontFamily}`;
		const canvas = document.createElement('canvas');
		const context = canvas.getContext('2d');
 
		if (!context) return;
		context.font = inputFont;
		const textWidth = context.measureText(this._input.value).width;
 
		this._isTruncated = textWidth > this._input.offsetWidth;
	}
 
	async firstUpdated() {
		await this.updateComplete;
		if (!this._input.value) return;
		lazyLoading(this, () => {
			this._checkInputOverflow();
			this._lastValue = this._input.value;
		});
	}
}
