import { LitElement, unsafeCSS } from 'lit';
import { property } from 'lit/decorators.js';
import { booleanType } from '../../utils/property-types';
import styles from './ng-textarea.style.css?inline';
import { template } from './ng-textarea.template';
 
export class NgTextarea extends LitElement {
	static get styles() {
		return unsafeCSS(styles);
	}

	@property({ type: String }) set value(value: string) {
		if (value !== undefined) this.requestUpdate();
	}

	get value() {
		if (this._textarea) {
			return this._textarea.value;
		}
		return '';
	}

	@property(booleanType) set showError(showError: boolean) {
		const oldValue = this._showError;
		this._showError = showError;
		this.requestUpdate('showError', oldValue);
	}

	@property(booleanType) autoHeight = false;

	get showError() {
		return this._showError;
	}

	get _label() {
		const label: HTMLSlotElement | undefined = this.shadowRoot?.querySelector('slot[name="label"]') || undefined;

		return label?.assignedElements()[0] as HTMLElement;
	}

	get _textarea() {
		const textarea: HTMLSlotElement | undefined = this.shadowRoot?.querySelector('slot[name="textarea"]') || undefined;

		return textarea?.assignedElements()[0] as HTMLTextAreaElement;
	}

	get _description() {
		const description: HTMLSlotElement | undefined =
			this.shadowRoot?.querySelector('slot[name="description"]') || undefined;

		return description?.assignedElements()[0] as HTMLElement;
	}

	@property({ type: String }) size = 'lg';

	@property({ type: String }) icon: string | undefined = undefined;

	_maxLength = 0;

	_isTextareaFocused = false;

	_isGroupFocusedVisible = false;

	_showError = false;

	// Options for the observer (which mutations to observe)
	observerConfig = { attributes: true, childList: true, subtree: true };

	// Callback function to execute when mutations are observed
	callback = (mutationList: any) => {
		for (const mutation of mutationList) {
			if (mutation.type === 'attributes') {
				this._checkAttributes();
			}
		}
		this.requestUpdate();
	};

	// Create an observer instance linked to the callback function
	observer = new MutationObserver(this.callback);

	private _handleInputBound = this._handleInput.bind(this);
	private _handleFocusBound = this._handleFocus.bind(this);
	private _handleFocusOutBound = this._handleFocusOut.bind(this);
	private _handleBlurBound = this._handleBlur.bind(this);
	private _handelLabelClickBound = this._handelLabelClick.bind(this);
	private _handleKeyupBound = this._handleKeyup.bind(this);
 
	disconnectedCallback() {
		this._textarea?.removeEventListener('input', this._handleInputBound);
		this._textarea?.removeEventListener('focus', this._handleFocusBound);
		this._textarea?.removeEventListener('focusout', this._handleFocusOutBound);
		this._textarea?.removeEventListener('blur', this._handleBlurBound);
		this._textarea?.removeEventListener('keyup', this._handleKeyupBound);
		this._label?.removeEventListener('click', this._handelLabelClickBound);
		this.observer.disconnect();
	}
 
	private _checkAttributes() {
		if (this._textarea) {
			const { maxLength } = this._textarea;
			this._maxLength = maxLength > 0 ? maxLength : 0;
		}
	}
 
	async firstUpdated() {
		try {
			await this.updateComplete;
			this._checkAttributes();
			this._addEventListeners();
			if (this.autoHeight) {
				this._updateHeight();
			}
			this.requestUpdate();
			this.observer.observe(this._textarea, this.observerConfig);
		} catch (e) {
			console.error('ERROR OCURRED', e);
		}
	}
 
	_addEventListeners() {
		this._textarea.addEventListener('input', this._handleInputBound);
		this._textarea.addEventListener('focus', this._handleFocusBound);
		this._textarea.addEventListener('focusout', this._handleFocusOutBound);
		this._textarea.addEventListener('blur', this._handleBlurBound);
		this._label?.addEventListener('click', this._handelLabelClickBound);
		this._textarea.addEventListener('keyup', this._handleKeyupBound);
	}
 
	_updateHeight() {
		this._textarea.style.height = 'auto';
		this._textarea.style.height = `${this._textarea.scrollHeight}px`;
	}
 
	_handleInput() {
		if (this.autoHeight) {
			this._updateHeight();
		}
		this._showError = !this._textarea.checkValidity();
		this.requestUpdate();
	}
 
	_handleKeyup(e: any) {
		const code = e.keyCode ? e.keyCode : e.which;
		if (code === 9) {
			this._handleFocus();
		}
	}
 
	_handleFocus() {
		this._isGroupFocusedVisible = true;
		this._isTextareaFocused = true;
		this.requestUpdate();
	}
 
	_handleFocusOut() {
		this._handleBlur();
		this._isGroupFocusedVisible = false;
		this._isTextareaFocused = false;
		this.requestUpdate();
	}
 
	_handleBlur() {
		this._isTextareaFocused = false;
		this.requestUpdate();
	}
 
	_handelLabelClick() {
		this._textarea.focus();
		this.requestUpdate();
	}
 
	_handleSlotChange() {
		this.requestUpdate();
	}
 
	render() {
		return template(this);
	}
}
