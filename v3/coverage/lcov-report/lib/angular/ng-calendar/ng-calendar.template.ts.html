import { classMap } from 'lit/directives/class-map.js';
import { html, literal, unsafeStatic } from 'lit/static-html.js';
import { getCustomElementSuffix } from '../../api/custom-element-scope';
import { MONTH, NgCalendar, WEEK } from './ng-calendar';
 
const dssNgTimepicker = literal`dss-ng-timepicker${unsafeStatic(getCustomElementSuffix())}`;
const dssIcon = literal`dss-icon${unsafeStatic(getCustomElementSuffix())}`;
const dssIconButton = literal`dss-icon-button${unsafeStatic(getCustomElementSuffix())}`;
const dssButton = literal`dss-button${unsafeStatic(getCustomElementSuffix())}`;
 
export const template = (component: NgCalendar) => html`
  <div
        class="calendar__container"
        @keydown="${component._handleCalendarKeydown}"
      >
        <div class="calendar__content">
          <div class="calendar__header">
            <div class="calendar__header-item calendar__header-selector">
              <div
                id="firstCalendarElement"
                class="calendar__header-item calendar__header-item--click calendar__header-month"
                tabindex="0"
                role="button"
                @keydown="${component._onHeaderMonthKeyDown}"
                @click="${component._toggleMonthSelector}"
                aria-label="selecciona el mes, actualment ${MONTH[component._currMonth]}"
              >
                ${MONTH[component._currMonth]}
                <${dssIcon} icon="expand_more" size="sm"></${dssIcon}>
              </div>
              <div
                class="calendar__header-item calendar__header-item--click calendar__header-year"
                tabindex="0"
                role="button"
                @keydown="${component._onHeaderYearKeyDown}"
                @click=${component._toggleYearSelector}
                aria-label="selecciona l'any, actualment ${component._currYear}"
              >
                ${component._currYear}
                <${dssIcon} icon="expand_more" size="sm"></${dssIcon}>
              </div>
            </div>
            <div class="calendar__header-item calendar__header-buttons">
              <${dssIconButton} hideTooltip variant="primary" icon="chevron_left" label="Anterior" @click=${component._prev}
                ?disabled=${
									component._currMonth === component._minDate?.getMonth() &&
									component._currYear === component._minDate?.getFullYear()
								}>
              </${dssIconButton}>
              <${dssIconButton} hideTooltip variant="primary" icon="chevron_right" label="Següent" @click=${component._next}
                ?disabled=${
									component._currMonth === component._minDate?.getMonth() &&
									component._currYear === component._minDate?.getFullYear()
								}>
              </${dssIconButton}>
            </div>
          </div>

          <div class="calendar__days-content">
            <ul class="calendar__week-names">
              ${WEEK.map((day: string) => html`<li>${day}</li>`)}
            </ul>
            <div
              class="calendar__days"
              @mouseleave=${component._removeRangeOverDate}
            >
              ${component._days.map((day: number) => {
								const cssClass = {
									'calendar__day-item--active': component._isToday(day),
									'calendar__day-item--weekend': component._isWeekend(day),
									'calendar__day-item--selected': component._isSelected(day),
									[`calendar__day-item--${component._getCustomType(day)}`]: Boolean(component._getCustomType(day)),
									'calendar__day-item--range':
										component._isBetweenRange(day) || component._isBetweenRangeOnMouseOver(day),
									'calendar__day-item--selected-range-start': component._isSelectedRangeStart(day),
									'calendar__day-item--selected-range-end':
										component._isSelectedRangeEnd(day) || component._isOverRangeDate(day),
									'calendar__day-item--range-enabled': component._range,
								};
								const handleDayKeydown = (event: KeyboardEvent) => {
									const day = event.target as HTMLButtonElement;

									if (event.key === 'ArrowUp') {
										event.preventDefault();
										const prevRowBtn = findVerticalDay(day, -7);
										if (prevRowBtn) {
											prevRowBtn.focus();
										} else {
											moveToPrevMonth();
										}
									}

									if (event.key === 'ArrowDown') {
										event.preventDefault();
										const nextRowBtn = findVerticalDay(day, 7);
										if (nextRowBtn) {
											nextRowBtn.focus();
										} else {
											moveToNextMonth();
										}
									}

									if (event.key === 'ArrowRight') {
										event.preventDefault();
										const nextDay: any = day.nextElementSibling;
										if (nextDay) {
											nextDay.focus();
										} else {
											moveToNextMonth();
										}
									}

									if (event.key === 'ArrowLeft') {
										event.preventDefault();
										const prevDay = findPreviousEnabledButton(day);
										if (prevDay) {
											prevDay.focus();
										} else {
											moveToPrevMonth();
										}
									}
								};

								const moveToPrevMonth = () => {
									component._prev();
									setTimeout(() => {
										const daysButtons: any = component.shadowRoot?.querySelectorAll(
											'.calendar__day-item:not(.calendar__day-item--empty)',
										);
										const enabledButtons = Array.from(daysButtons).filter((btn: any) => !btn.disabled);
										const lastDayOfPrevMonth: any = enabledButtons[enabledButtons.length - 1];
										if (lastDayOfPrevMonth) {
											lastDayOfPrevMonth.setAttribute('tabindex', '0');
											lastDayOfPrevMonth.focus();
										}
									}, 0);
								};

								const moveToNextMonth = () => {
									component._next();
									setTimeout(() => {
										const daysButtons: any = component.shadowRoot?.querySelectorAll(
											'.calendar__day-item:not(.calendar__day-item--empty)',
										);
										const enabledButtons = Array.from(daysButtons).filter((btn: any) => !btn.disabled);
										const firstDayOfNextMonth: any = enabledButtons[0];
										if (firstDayOfNextMonth) {
											firstDayOfNextMonth.setAttribute('tabindex', '0');
											firstDayOfNextMonth.focus();
										}
									}, 0);
								};

								const findPreviousEnabledButton = (start: HTMLElement | null): HTMLElement | null => {
									let current = start?.previousElementSibling as HTMLButtonElement | null;
									while (current) {
										if (
											!current.classList.contains('calendar__day-item') ||
											current.classList.contains('calendar__day-item--empty')
										) {
											current = current.previousElementSibling as HTMLButtonElement | null;
											continue;
										}
										if (!current.disabled) {
											return current;
										}
										current = current.previousElementSibling as HTMLButtonElement | null;
									}
									return null;
								};

								const findVerticalDay: any = (
									current: HTMLButtonElement,
									offset: number, // +7 para ↓, -7 para ↑
								) => {
									const allButtons = Array.from(
										current
											.closest('.calendar__days')
											?.querySelectorAll('.calendar__day-item:not(.calendar__day-item--empty)') || [],
									) as HTMLButtonElement[];
									const enabledButtons = allButtons.filter((btn) => !btn.disabled);
									const currentIndex = enabledButtons.indexOf(current);
									const targetIndex = currentIndex + offset;
									if (targetIndex < 0 || targetIndex >= enabledButtons.length) {
										return null;
									}
									return enabledButtons[targetIndex];
								};

								return html`
                  <button
                    tabindex="${component._isFocusable(day) ? 0 : -1}"
                    class="calendar__day-item ${classMap(cssClass)}"
                    @click=${() => component._selectDate(day)}
                    @mouseover=${() => component._selectRangeOverDate(day)}
                    @focus=${() => component._selectRangeOverDate(day)}
                    ?disabled=${component._isInactive(day)}
                    @keydown="${handleDayKeydown}"
                  >
                    ${day || null}
                  </button>
                `;
							})}
            </div>
          </div>
        </div>
        ${
					component._showMonthSelector
						? html`
              <div class="calendar-selector calendar-selector--month">
                <div class="calendar-selector-title">Selecciona un mes</div>
                <div class="calendar-selector-options">
                  ${component._generateMonthsOptions()}
                </div>
              </div>
            `
						: null
				}
        ${
					component._showYearSelector
						? html`
              <div class="calendar-selector calendar-selector--year">
                <div
                  class="calendar-selector-title calendar-selector-title--years"
                >
                  <div class="calendar-selector-title__years">
                    ${component._yearsRangeStart} - ${component._yearsRangeEnd}
                  </div>
                  <div class="calendar-selector-title__actions">
                    <${dssIconButton}
                      hideTooltip
                      variant="primary" 
                      label="Anys anteriors"
                      icon="chevron_left"
                      @click=${component._onYearRangeStepDown}
                    ></${dssIconButton}>
                    <${dssIconButton}
                      hideTooltip
                      variant="primary" 
                      label="Anys posteriors"
                      icon="chevron_right"
                      @click=${component._onYearRangeStepUp}
                    ></${dssIconButton}>
                  </div>
                </div>
                <div
                  class="calendar-selector-options calendar-selector-options--4col"
                >
                  ${component._generateYearsRangeOptions()}
                </div>
              </div>
            `
						: null
				}
        ${
					component._showTime
						? html`
              <div class="dss-datepicker__timepicker">
                <${dssNgTimepicker}
                  .dropdown="${component._timepicker}"
                  .customTimeListOptions=${component._customTimeListOptions}
                  .minutesRange=${component._minutesRange}
                  .minHour=${component._minHour}
                  .maxHour=${component._maxHour}
                  @onTimepickerChange=${component._onSelectTime}
                  .value="${component._timepickerValue}"
                >
                  <label slot="label" for="innerTimepicker"
                    >${component._timepickerLabel}</label
                  >
                  <input
                    slot="input"
                    id="innerTimepicker"
                    type="text"
                    maxlength="5"
                    .value="${component._timepickerValue}"
                  />
                </${dssNgTimepicker}>
              </div>
            `
						: null
				}
        ${
					component._showButtons || component._showTime
						? html`
              <div class="calendar__content calendar__content--buttons">
                <div class="calendar__buttons">
                  <${dssButton}
                    variant="secondary"
                    label=${component._leftLabel}
                    @click=${component._onCancel}
                  >
                  </${dssButton}>
                  <${dssButton}
                    variant="primary"
                    label=${component._rightLabel}

                    @click=${component._onAccept}
                    ?disabled=${component._validateSelectedDate()}
                  >
                  </${dssButton}>
                </div>
              </div>
            `
						: null
				}
      </div>
 
`;
