import { Placement, autoUpdate, computePosition, flip, offset, shift } from '@floating-ui/dom';
 
import { LitElement, unsafeCSS } from 'lit';
import { property } from 'lit/decorators.js';
import resetStyles from '../../shared/reset.style.css?inline';
import { booleanType } from '../../utils/property-types';
import styles from './action-menu.style.css?inline';
import { actionMenuTemplate } from './action-menu.template';
 
export class ActionMenu extends LitElement {
	static get styles() {
		return [unsafeCSS(resetStyles), unsafeCSS(styles)];
	}
 
	private _parentMouseDownHandler: ((event: MouseEvent) => void) | null = null;
	private _parentKeyDownHandler: ((event: KeyboardEvent) => void) | null = null;
	private _slotChangeHandler: (() => void) | null = null;
 
	constructor() {
		super();
		this._handleOutsideClick = this._handleOutsideClick.bind(this);
		this._handleFocusOut = this._handleFocusOut.bind(this);
	}
 
	connectedCallback() {
		super.connectedCallback();
		this.addEventListener('item-close', this._handleCloseAllMenus);
		this._handleConnectedCallback();
	}
 
	disconnectedCallback() {
		this._removeDropdownListener();
		this.removeEventListener('item-close', this._handleCloseAllMenus);
		this.visibleObserver.disconnect();
		this._cleanupFloating?.();
		this._cleanupFloating = null;
	}
 
	_addDropdownListener() {
		document.addEventListener('mousedown', this._handleOutsideClick);
		this.addEventListener('focusout', this._handleFocusOut);
	}
 
	_removeDropdownListener() {
		document.removeEventListener('mousedown', this._handleOutsideClick);
		this.removeEventListener('focusout', this._handleFocusOut);
	}
 
	_handleOutsideClick(event: any) {
		this._checkClickOutside(event);
	}
 
	_handleFocusOut(event: FocusEvent) {
		this._checkFocusOut(event);
	}
 
	_handleConnectedCallback() {
		if (this._isFirstUpdate) return;
		const position = this.style.position;
		if (position !== 'absolute' && position !== 'fixed' && position !== 'relative') {
			if (this._referenceEl) {
				this._updateFloatingPosition();
			} else {
				this._initActionMenu();
			}
		}
	}
 
	private visibleObserver = new IntersectionObserver(
		([entry]) => {
			if (!entry.isIntersecting) {
				this.classList.remove('visible');
				this._isVisible = false;
				this._removeDropdownListener();
			}
		},
		{
			root: null,
			threshold: 0,
		},
	);
 
	@property(booleanType) fullWidth = false;
 
	@property({ type: String }) position = 'right-start';
 
	@property(booleanType) disablePopper = false;
 
	@property(booleanType) disableParentClick = false;
 
	@property(booleanType) open = false;
 
	@property(booleanType) dropdownFixed = false;
 
	_cleanupFloating: (() => void) | null = null;
 
	_referenceEl: HTMLElement | null = null;
 
	_floatingEl: HTMLElement | null = null;
 
	_parent: HTMLElement | null = null;
 
	_trigger: HTMLElement | null = null;
 
	_isVisible = false;
 
	_disableClickOutside = true;
 
	_isFirstUpdate = true;
 
	_createFloatingInstance(parent: HTMLElement) {
		this._referenceEl = parent;
		this._floatingEl = this;
 
		// Limpia si existía
		this._cleanupFloating?.();
 
		this._cleanupFloating = autoUpdate(this._referenceEl, this._floatingEl, () => this._updateFloatingPosition());
 
		this.visibleObserver.observe(parent);
 
		this._parentMouseDownHandler = () => this._openByParent();
		this._parentKeyDownHandler = (event: KeyboardEvent) => {
			const key = event.key;
			if (key === 'Enter' || key === ' ') {
				this._openByParent();
			}
		};
 
		parent.addEventListener('mousedown', this._parentMouseDownHandler);
		parent.addEventListener('keydown', this._parentKeyDownHandler);
	}
 
	async _updateFloatingPosition() {
		if (!this._referenceEl || !this._floatingEl) return;
 
		const { x, y, strategy } = await computePosition(this._referenceEl, this._floatingEl, {
			placement: this.position as Placement,
			strategy: this.dropdownFixed ? 'fixed' : 'absolute',
			middleware: [
				offset(8),
				flip(),
				shift({
					padding: { top: 8, bottom: 8, left: 16, right: 16 },
				}),
			],
		});
 
		Object.assign(this._floatingEl.style, {
			position: strategy,
			left: `${x}px`,
			top: `${y}px`,
			transform: 'none',
		});
	}
 
	private _openByParent() {
		if (!this._isVisible && !this.disableParentClick) {
			setTimeout(() => {
				this._updateFloatingPosition();
				this.classList.add('visible');
				this._disableClickOutside = false;
				this._isVisible = true;
				this._addDropdownListener();
			}, 100);
		}
	}
 
	private _handleCloseAllMenus() {
		this._closeMenu();
	}
 
	private _handleSlotChange() {
		const slot = this.shadowRoot?.querySelector('slot');
		if (slot) {
			const elements = slot
				.assignedElements({ flatten: true })
				.filter((el) => el.tagName.toLowerCase() === 'dss-action-menu-item');
			elements.forEach((el, index) => {
				if (index === 0) {
					el.setAttribute('first', '');
				} else {
					el.removeAttribute('first');
				}
				if (index === elements.length - 1) {
					el.setAttribute('last', '');
				} else {
					el.removeAttribute('last');
				}
			});
		}
	}
 
	private _checkClickOutside(event: Event) {
		if (this.disableParentClick && this._disableClickOutside) {
			return;
		}
 
		const path = event.composedPath();
 
		const selfTriggerItem = this.closest('dss-action-menu-item');
 
		const isClickInSelfTrigger = selfTriggerItem && path.includes(selfTriggerItem as HTMLElement);
 
		if (selfTriggerItem && !isClickInSelfTrigger) {
			return this._closeMenu();
		}
 
		const isClickInParent = path.includes(this._parent as HTMLElement);
 
		const isClickInsideMenu = path.some(
			(el) =>
				el instanceof HTMLElement &&
				(el.tagName.toLowerCase() === 'dss-action-menu' ||
					(el.tagName.toLowerCase() === 'ul' && el.classList.contains('dss-action-menu'))),
		);
 
		if (!isClickInsideMenu && !isClickInParent && !this.disablePopper) {
			return this._closeMenu();
		}
 
		// FIX: split button not closing because it's parent at same time
		const isParentSplitButton = (this._parent as HTMLElement).classList.contains('dss-split-button-action');
		const openBySplitButton = this._parent?.getAttribute('data-expanded'); // typeof string: true or false
		if (isParentSplitButton && openBySplitButton === 'false') {
			return this._closeMenu();
		}
	}
 
	private _checkFocusOut(event: FocusEvent) {
		const relatedTarget = event.relatedTarget as HTMLElement;
		if (
			relatedTarget !== null &&
			relatedTarget !== this &&
			relatedTarget.tagName.toLowerCase() !== 'dss-action-menu-item'
		) {
			this._closeMenu();
		}
	}
 
	_closeMenu() {
		setTimeout(() => {
			if (this.classList.contains('visible')) {
				this.classList.remove('visible');
				this._isVisible = false;
				this.dispatchEvent(new CustomEvent('close'));
				this._removeDropdownListener();
			}
		}, 0);
	}
 
	_initActionMenu() {
		// Check if the component is slotted,
		// If not, parent is default HTMLElement parentElement
		const assignedSlot: any = this.assignedSlot;
		this._parent = assignedSlot ? assignedSlot.parentElement : this.parentElement;
		if (this._parent && !this.disablePopper) {
			this._createFloatingInstance(this._parent);
		}
 
		if (this.disablePopper) {
			this.classList.add('visible');
			this._addDropdownListener();
		}
 
		const slot = this.shadowRoot?.querySelector('slot');
		if (slot) {
			this._slotChangeHandler = () => this._handleSlotChange();
			slot.addEventListener('slotchange', this._slotChangeHandler);
		}
		this._handleSlotChange(); // Initial call to handle already slotted elements
	}
 
	async firstUpdated() {
		await this.updateComplete;
		this._initActionMenu();
		this._isFirstUpdate = false;
	}
 
	updated(changedProperties: Map<string | number | symbol, unknown>) {
		super.updated(changedProperties);
		if (changedProperties.has('position')) {
			this._updateFloatingPosition();
		}
		if (changedProperties.has('open')) {
			if (this.open) {
				this.classList.add('visible');
				this._isVisible = true;
				this._updateFloatingPosition();
				setTimeout(() => {
					this._addDropdownListener();
					this._disableClickOutside = false;
				}, 100);
			} else {
				this.classList.remove('visible');
				this._isVisible = false;
				this._disableClickOutside = true;
				this._removeDropdownListener();
			}
		}
	}
 
	render() {
		return actionMenuTemplate(this);
	}
}
