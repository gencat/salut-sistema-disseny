import { LitElement, unsafeCSS } from 'lit';
import { property, state } from 'lit/decorators.js';
import { booleanType } from '../../utils/property-types';
import styles from './slider.style.css?inline';
import { template } from './slider.template';
 
export class Slider extends LitElement {
	static get styles() {
		return unsafeCSS(styles);
	}
 
	private _handleDomContentLoaded: () => void;
	private _handleInputEvent: () => void;
	private _handleMouseUp: () => void;
	private _handleKeyUp: () => void;
	private _handleTouchStartEvent: (event: Event) => void;
	private _handleTouchMoveEvent: (event: Event) => void;
	private _handleTouchEndEvent: (event: Event) => void;
 
	get _input() {
		return this.shadowRoot?.querySelector('input') as HTMLInputElement;
	}
 
	get _tooltip() {
		return this.shadowRoot?.querySelector(`#dss-slider-tooltip`) as HTMLElement;
	}
 
	@property({ type: String }) label = 'Input range';
 
	@property({ type: Number }) value = 0;
 
	@property({ type: Number }) min = 0;
 
	@property({ type: Number }) max = 0;
 
	@property({ type: Number }) step = 1;
 
	@property({ type: String }) orientation = 'horizontal';
 
	@property(booleanType) disabled = false;
 
	@state() _progress = 0;
 
	@state() _vertical = false;
 
	@state() _isTooltipTouched = false;
 
	// Deprecated state, use status instead
	@property({ type: String }) set orient(orient: string) {
		this.orientation = orient;
	}
 
	get orient() {
		return this.orientation;
	}
 
	updated(changedProperties: any) {
		if (changedProperties.has('value') || changedProperties.has('max')) {
			queueMicrotask(() => {
				if (this.max >= this.value) {
					this._setProgress();
				}
			});
		}
 
		if (changedProperties.has('orientation')) {
			queueMicrotask(() => {
				this._vertical = this.orientation === 'vertical';
			});
		}
	}
 
	constructor() {
		super();
		this._handleDomContentLoaded = this._updateTooltip.bind(this);
		this._handleInputEvent = this._updateTooltip.bind(this);
		this._handleMouseUp = this._dispatchValue.bind(this);
		this._handleKeyUp = this._dispatchValue.bind(this);
		this._handleTouchStartEvent = this._handleTouchStart.bind(this);
		this._handleTouchMoveEvent = this._handleTouchMove.bind(this);
		this._handleTouchEndEvent = this._handleTouchEnd.bind(this);
	}
 
	async firstUpdated() {
		try {
			await this.updateComplete;
			this._setProgress();
			document.addEventListener('DOMContentLoaded', this._handleDomContentLoaded);
			this._input.addEventListener('input', this._handleInputEvent);
			this._input.addEventListener('mouseup', this._handleMouseUp); // Return new input value after dragging
			this._input.addEventListener('keyup', this._handleKeyUp); // Return new input value after dragging
			this._input.addEventListener('touchstart', this._handleTouchStartEvent); // Return new input value after clicking
			this._input.addEventListener('touchmove', this._handleTouchMoveEvent); // Return new input value after clicking
			this._input.addEventListener('touchend', this._handleTouchEndEvent); // Return new input value after clicking
		} catch (e) {
			console.error('ERROR OCURRED');
		}
	}
 
	disconnectedCallback() {
		super.disconnectedCallback();
		document.removeEventListener('DOMContentLoaded', this._handleDomContentLoaded);
		this._input?.removeEventListener('input', this._handleInputEvent);
		this._input?.removeEventListener('mouseup', this._handleMouseUp);
		this._input?.removeEventListener('keyup', this._handleKeyUp);
		this._input?.removeEventListener('touchstart', this._handleTouchStartEvent);
		this._input?.removeEventListener('touchmove', this._handleTouchMoveEvent);
		this._input?.removeEventListener('touchend', this._handleTouchEndEvent);
	}
 
	_handleTouchStart(event: any) {
		if (this._vertical) {
			event.preventDefault();
		}
		this._isTooltipTouched = true;
	}
 
	_handleTouchEnd(event: any) {
		if (this._vertical) {
			event.preventDefault();
		}
		this._isTooltipTouched = false;
		this._dispatchValue();
	}
 
	_handleTouchMove(event: any) {
		if (this._vertical) {
			if (!event.target) return;
			const slider = event.target as HTMLInputElement;
			const max = Number.parseInt(slider.max, 10);
			const min = Number.parseInt(slider.min, 10);
 
			const touchLocation = event.touches[0];
			const sliderRect = slider.getBoundingClientRect();
			const touchPercentage = (touchLocation.clientY - sliderRect.top) / sliderRect.height;
			const touchValue = Math.round(touchPercentage * (max - min) + min);
 
			slider.value = String(max - touchValue + min);
			slider.dispatchEvent(new Event('input'));
		}
	}
 
	_handleInput() {
		if (this._input) {
			const tempValue = Number.parseFloat(this._input.value);
			this._progress = Math.round(((tempValue - this.min) / (this.max - this.min)) * 100);
		}
	}
 
	_setProgress() {
		if (this._input) {
			this._input.value = this.value.toString();
		}
		this._progress = Math.round(((this.value - this.min) / (this.max - this.min)) * 100);
	}
 
	_updateTooltip() {
		if (this._tooltip) {
			const newValue = Number(((+this._input.value - +this.min) * 100) / (+this.max - +this.min));
			const newPosition = 10 - newValue * 0.2;
 
			this._tooltip.innerHTML = `<span class="tooltip-wrapper"><span class="tooltip-value">${this._input.value}</span></span>`;
			this._tooltip.style.left = `calc(${newValue}% + (${newPosition}px))`;
		}
	}
 
	_dispatchValue() {
		const options = {
			detail: this._input.value,
			bubbles: true,
			composed: true,
		};
		this.dispatchEvent(new CustomEvent('value-changed', options));
	}
 
	render() {
		return template(this);
	}
}
