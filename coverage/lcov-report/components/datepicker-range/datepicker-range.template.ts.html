import { nothing } from 'lit';
import { classMap } from 'lit/directives/class-map.js';
import { html, literal, unsafeStatic } from 'lit/static-html.js';
import { getCustomElementSuffix } from '../../api/custom-element-scope';
import { DatepickerRange } from './datepicker-range';
 
const dssCalendar = literal`dss-calendar${unsafeStatic(getCustomElementSuffix())}`;
const dssIcon = literal`dss-icon${unsafeStatic(getCustomElementSuffix())}`;
const dssIconButton = literal`dss-icon-button${unsafeStatic(getCustomElementSuffix())}`;
 
export const template = (component: DatepickerRange) => {
	const datepickerRangeClass = {
		'dss-datepicker-range--sm': component.size !== 'lg',
	};
 
	const helpClass = {
		'dss-datepicker-range-help--invalid':
			component.invalid ||
			(!component._inputRangeStart?.validity.valid && component._inputRangeStart?.value !== '') ||
			(!component._inputRangeEnd?.validity.valid && component._inputRangeEnd?.value !== ''),
		'dss-datepicker-range-help--disabled': component._inputRangeStart?.disabled && component._inputRangeEnd?.disabled,
	};
 
	const startWrapperClassmap = {
		'dss-input-wrapper': true,
		'dss-input-wrapper--required': component._inputRangeStart?.required,
		'dss-input-wrapper--disabled': component._inputRangeStart?.disabled,
		[`dss-input-wrapper--${component.size}`]: Boolean(component.size),
	};
 
	const endWrapperClassmap = {
		'dss-input-wrapper': true,
		'dss-input-wrapper--required': component._inputRangeEnd?.required,
		'dss-input-wrapper--disabled': component._inputRangeEnd?.disabled,
		[`dss-input-wrapper--${component.size}`]: Boolean(component.size),
	};
 
	const startInputGroupClass = {
		'dss-input-group': true,
		[`dss-input-group--${component.size}`]: Boolean(component.size),
		'dss-input-group--invalid':
			component.invalid || (!component._inputRangeStart?.validity.valid && component._inputRangeStart?.value !== ''),
		'dss-input-group--required': component._inputRangeStart?.required,
		'dss-input-group--disabled': component._inputRangeStart?.disabled,
		'dss-input-group--focused':
			component._inputRangeStart?.value || component._isStartFocused || component._copyInputRangeStartPlaceholder,
		'dss-input-group--read-only': component._inputRangeStart?.readOnly,
		// 'dss-input-group--no-label': Boolean(!component._label),
	};
 
	const endInputGroupClass = {
		'dss-input-group': true,
		[`dss-input-group--${component.size}`]: Boolean(component.size),
		'dss-input-group--invalid':
			component.invalid || (!component._inputRangeEnd?.validity.valid && component._inputRangeEnd?.value !== ''),
		'dss-input-group--required': component._inputRangeEnd?.required,
		'dss-input-group--disabled': component._inputRangeEnd?.disabled,
		'dss-input-group--focused':
			component._inputRangeEnd?.value || component._isEndFocused || component._copyInputRangeEndPlaceholder,
		'dss-input-group--read-only': component._inputRangeEnd?.readOnly,
		// 'dss-input-group--no-label': Boolean(!component._label),
	};
 
	// const calendarClass = {
	// 	'dss-calendar': true,
	// 	'dss-calendar--visible':
	// 		component.showCalendar && !component._inputRangeStart?.readOnly && !component._inputRangeEnd?.readOnly,
	// 	'dss-calendar--disabled': component._inputRangeStart?.disabled && component._inputRangeEnd?.disabled,
	// 	'dss-calendar--sm': component.size !== 'lg',
	// };
 
	return html`
      <div class="dss-datepicker-range ${classMap(datepickerRangeClass)}">
        <input 
          type="hidden" 
          class="dss-hidden-value" 
          .name=${component.name}
        />
 
        <div 
          class="dss-datepicker-range-inputs">
 
          <div class="dss-datepicker-range-inputs__start ${classMap(startWrapperClassmap)}">
            ${
							component.size === 'sm'
								? html`
                <div class="${classMap({
									'dss-wrapper-label': true,
									'dss-wrapper-label--invalid': component.invalid,
								})}">
                  <label class="dss-label dss-label--start" for="${component._getEffectiveStartId()}">${component.labelRangeStart}</label>
                </div>
              `
								: nothing
						}
            <div class="${classMap(startInputGroupClass)}">
              ${
								component.iconRangeStart && component.iconRangeStart !== ''
									? html`
                <${dssIcon} icon="${component.iconRangeStart}" class="dss-input-icon"></${dssIcon}>
                `
									: nothing
							}
              <div class="dss-input-field">
                ${
									component.size !== 'sm' && !component.hideLabel
										? html`
                      <label class="dss-label dss-label--start" for="${component._getEffectiveStartId()}">${component.labelRangeStart}</label>
                    `
										: nothing
								}
        
                <input
                  class="dss-input dss-input--start"
                  id=${component._getEffectiveStartId()}
                  .type=${component.type}
                  .name="${component.name ? `${component.name}-start` : nothing}"
                  .placeholder=${component.placeholderRangeStart}
                  ?disabled=${component.disabled}
                  ?readonly=${component.readonly}
                  ?required=${component.required}
                  ?autofocus=${component.autofocus}
                  spellcheck=${component.spellcheck ? 'true' : 'false'}
                  autocorrect=${component.autocorrect ? 'on' : 'off'}
                  autocomplete=${component.autocomplete}
                  autocapitalize=${component.autocapitalize}
                  pattern=${component.pattern ?? nothing}
                  inputmode=${component.inputmode ?? nothing}
                  aria-label="${component.hideLabel ? component.labelRangeStart : nothing}"
                  @click=${component._handleRangeStartClick}
                  @input=${component._handleRangeStartInput}
                  @focusin=${component._handleRangeStartFocusIn}
                  @keydown=${component._handleRangeKeydown}
                  role="combobox"
                  aria-haspopup="dialog"
                  aria-controls="${component.showCalendar ? `${component._getEffectiveId()}-calendar` : nothing}"
                  aria-expanded=${component.showCalendar ? 'true' : 'false'}
                />
              </div>
              <${dssIconButton}
                hideTooltip
                size="md"
                icon="close"
                label="Netejar"
                @onClick=${() => component._clearDate('rangeStart')}
                ?hidden=${!component._inputRangeStart?.value || component._inputRangeStart.readOnly || component._inputRangeStart.disabled}
              ></${dssIconButton}>
            </div>
          </div>
 
          <div class="dss-datepicker-range-inputs__end ${classMap(endWrapperClassmap)}">
            ${
							component.size === 'sm'
								? html`
                 <div class="${classMap({
										'dss-wrapper-label': true,
										'dss-wrapper-label--invalid': component.invalid,
									})}">
                  <label class="dss-label dss-label--end" for="${component._getEffectiveEndId()}">${component.labelRangeEnd}</label>
                </div>
              `
								: nothing
						}
            <div class="${classMap(endInputGroupClass)}">
              ${
								component.iconRangeEnd && component.iconRangeEnd !== ''
									? html`
                <${dssIcon} icon="${component.iconRangeEnd}" class="dss-input-icon"></${dssIcon}>
                `
									: nothing
							}
              <div class="dss-input-field">
                ${
									component.size !== 'sm' && !component.hideLabel
										? html`
                      <label class="dss-label dss-label--end" for="${component._getEffectiveEndId()}">${component.labelRangeEnd}</label>
                    `
										: nothing
								}
        
                <input
                  class="dss-input dss-input--end"
                  id=${component._getEffectiveEndId()}
                  .type=${component.type}
                  .name="${component.name ? `${component.name}-end` : nothing}"
                  .placeholder=${component.placeholderRangeEnd}
                  ?disabled=${component.disabled}
                  ?readonly=${component.readonly}
                  ?required=${component.required}
                  ?autofocus=${component.autofocus}
                  spellcheck=${component.spellcheck ? 'true' : 'false'}
                  autocorrect=${component.autocorrect ? 'on' : 'off'}
                  autocomplete=${component.autocomplete}
                  autocapitalize=${component.autocapitalize}
                  pattern=${component.pattern ?? nothing}
                  inputmode=${component.inputmode ?? nothing}
                  aria-label="${component.hideLabel ? component.labelRangeEnd : nothing}"
                  @click=${component._handleRangeEndClick}
                  @input=${component._handleRangeEndInput}
                  @focusin=${component._handleRangeEndFocusIn}
                  @keydown=${component._handleRangeKeydown}
                  role="combobox"
                  aria-haspopup="dialog"
                  aria-controls="${component.showCalendar ? `${component._getEffectiveId()}-calendar` : nothing}"
                  aria-expanded=${component.showCalendar ? 'true' : 'false'}
                />
              </div>
              <${dssIconButton}
                hideTooltip
                size="md"
                icon="close"
                label="Netejar"
                @onClick=${() => component._clearDate('rangeEnd')}
                ?hidden=${!component._inputRangeEnd?.value || component._inputRangeEnd.readOnly || component._inputRangeEnd.disabled}
              ></${dssIconButton}>
            </div>
          </div>
 
          <${dssCalendar}
            range
            .isRangeStartFocused=${component._isStartFocused}
            .isRangeEndFocused=${component._isEndFocused}
            id="${component._getEffectiveId()}-calendar"
            class="${classMap({
							[`dss-calendar ${component._getEffectiveId()}-calendar`]: true,
							'dss-calendar--md': component.size !== 'lg',
						})}"
            placement='bottom-start'
            .open=${component.showCalendar && !component.readonly}
            .selectedDate="${component._inputRangeStart?.value}"
            .rangeStartDate="${component._inputRangeStart?.value}"
            .rangeEndDate="${component._inputRangeEnd?.value}"
            .customCalendar=${component.customCalendar}
            .showButtons=${component.showButtons}
            .leftLabel=${component.calendarLeftButtonLabel}
            .rightLabel=${component.calendarRightButtonLabel}
            .minDate=${component.minDate}
            .maxDate=${component.maxDate}
            @range-changed=${component._onCalendarChange}
            @cancel=${component._onCalendarCancel}
            @calendar-focusout=${component._handleCalendarFocusOut}
          ></${dssCalendar}>
       
        </div>
 
        ${
					component._helpText
						? html`
              <div class="dss-datepicker-range-help ${classMap(helpClass)}">
                ${component._helpText}
              </div>
            `
						: null
				}
       
        
      </div>
    `;
};
